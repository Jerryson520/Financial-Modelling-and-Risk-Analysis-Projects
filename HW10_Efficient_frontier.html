# Define a function to compute the efficient frontier
def efficient_frontier(close_returns, num_portfolios, portfolio):
    # Define bounds and constraints
    bounds = tuple((0, 1) for _ in portfolio)
    constraints = ({'type': 'eq', 'fun': lambda x: np.sum(x) - 1})

    # Define an empty list to store the efficient frontier portfolios
    efficient_frontier = []

    # Calculate mean returns and covariance matrix
    mean_returns = close_returns.mean()
    cov_matrix = close_returns.cov()

    # Function to calculate portfolio standard deviation
    def portfolio_std(weights, cov_matrix):
        return np.sqrt(weights.T @ cov_matrix @ weights)

    # Minimize function for standard deviation given a target return
    for target_return in np.linspace(mean_returns.min(), mean_returns.max(), num_portfolios):
        args = (cov_matrix,)

        constraints = [{'type': 'eq', 'fun': lambda x: np.sum(x) - 1},
                       {'type': 'eq', 'fun': lambda x: np.dot(x, mean_returns) - target_return}]

        result = minimize(portfolio_std, len(portfolio) * [1. / len(portfolio)], args=args, method='SLSQP', bounds=bounds, constraints=constraints)

        if result.success:
            std = portfolio_std(result.x, cov_matrix)
            efficient_frontier.append((std, target_return))

    # Extract the risk (std_dev) and return for efficient frontier portfolios
    ef_risks, ef_returns = zip(*efficient_frontier)

    return ef_risks, ef_returns
